// Sample string let sampleString = " Hello, World! "; // 1. String length console.log("Length of string:", sampleString.length); // 2. String charAt() console.log("Character at index 4:", sampleString.charAt(4)); // 3. String slice() console.log("Slice from index 3 to 8:", sampleString.slice(3, 8)); // 4. String substring() console.log("Substring from index 3 to 8:", sampleString.substring(3, 8)); // 5. String substr() (Note: This is deprecated, use substring or slice instead) console.log("Substr from index 3, length 5:", sampleString.substr(3, 5)); // 6. String toUpperCase() console.log("Uppercase:", sampleString.toUpperCase()); // 7. String toLowerCase() console.log("Lowercase:", sampleString.toLowerCase()); // 8. String concat() let anotherString = " Welcome to JavaScript!"; console.log("Concatenated string:", sampleString.concat(anotherString)); // 9. String trim() console.log("Trimmed string:", sampleString.trim()); // 10. String trimStart() console.log("Trimmed start:", sampleString.trimStart()); // 11. String trimEnd() console.log("Trimmed end:", sampleString.trimEnd()); // 12. String repeat() console.log("Repeated string (3 times):", sampleString.trim().repeat(3)); // 13. String replace() console.log("Replace 'World' with 'JavaScript':", sampleString.replace("World", "JavaScript")); // 14. String replaceAll() console.log("Replace all 'o' with '0':", sampleString.replaceAll("o", "0")); // 15. String split() console.log("Split by space:", sampleString.trim().split(" "));
